import { Canvas, Meta, Story } from '@storybook/addon-docs';

import { List } from '../..';

<Meta title="Components/List/Styling" component={List} />

export const ListUnorderedTemplate = (args) => {
  return (
    <List {...args}>
      <li>element 1</li>
      <li>element 2</li>
      <li>element 3</li>
    </List>
  );
};
export const ListConcatTemplate = (args) => {
  return (
    <List>
      <li>element 1</li>
      <List {...args}>
        <li>element 1.1</li>
        <li>element 1.2</li>
        <li>element 1.3</li>
      </List>
    </List>
  );
};
export const ListDefinitionTemplate = (args) => {
  return (
    <List type="dl" {...args}>
      <dt>Cat</dt>
      <dd>Small furry feline</dd>
      <dt>Dog</dt>
      <dd>Large furry canine</dd>
    </List>
  );
};

# List

## Unstyled

Use `unstyled` to remove the bullets or numbers from a list

  <Story name="Unstyled" args={{ unstyled: true }}>
    {ListUnorderedTemplate.bind({})}
  </Story>
</Canvas>

## None

Use `none` to remove the bullets or numbers from a nested listed (maintains indenting)

  <Story name="None" args={{ none: true }}>

</Canvas>

## Spaced

Add `spaced` to increase the white space between the items, and make them visually separate and distinct
