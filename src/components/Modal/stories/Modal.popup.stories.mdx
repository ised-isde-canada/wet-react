<Meta title="Components/Modal/Pop-Up Modal" component={Modal.Dialog} />

export const TemplateGroup = (args) => {
  const [showModal, setShowModal] = React.useState(false);
  return (
    <>
      <Button onClick={() => setShowModal(true)}>Popup Modal Button</Button>
      <Modal
        show={showModal}
        animation={false}
        centered
        onHide={() => setShowModal(false)}
        {...args}
      >
        <Modal.Header closeButton closeLabel="goodbye">
          <Modal.Title> {args.title} </Modal.Title>
        </Modal.Header>
        <Modal.Body>{args.content} </Modal.Body>
        <Modal.Footer> {args.footer}</Modal.Footer>
      </Modal>
    </>
  );
};

```
    <Modal

      ...
    </Modal>
  </>

```
