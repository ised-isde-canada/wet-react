import { ArgsTable, Canvas, Meta, Story } from '@storybook/addon-docs';

import { MultimediaPlayer } from '../..';

import puppies from '@storybook/react/puppies.mp4';
import valkyries from '@storybook/react/ride_of_the_valkyries.mp3';

export const vidTemplate = (args) => (
  <MultimediaPlayer
    sources={[
      {
        type: 'mp4',
        source: puppies,
      },
    ]}
    {...args}
  />
);

export const audTemplate = (args) => (
  <MultimediaPlayer
    sources={[
      {
        type: 'mp3',
        source: valkyries,
      },
    ]}
    {...args}
  />
);

export const youTemplate = (args) => (
  <MultimediaPlayer
    sources={[
      {
        type: 'youtube',
        source: 'https://www.youtube.com/embed/a3ICNMQW7Ok',
      },
    ]}
    {...args}
  />
);

<Meta
  title="Components/MultimediaPlayer/Share/URL"

/>

# MultimediaPlayer

## shareURL

**Optional:** Add a "Share this video" or "Share this audio file" link by adding a `share` attribute comprised of a URL string

## Example Video URL

<Canvas>
  <Story
    name="Video URL"
    args={{ share: 'https://www.youtube.com/watch?v=YLslsZuEaNE' }}
  >
    {vidTemplate.bind({})}
  </Story>
</Canvas>

<Canvas>
