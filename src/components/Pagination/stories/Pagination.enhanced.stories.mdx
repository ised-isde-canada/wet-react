import { Canvas, Meta, Story } from '@storybook/addon-docs';

import { Pagination } from '../..';
import { PaginationButton } from '../..';

<Meta
  title="Components/Pagination/Enhanced"
  component={Pagination}
  subcomponents={{ 'Pagination.Button': Pagination.Button }}
/>

export const PaginationActiveTemplate = (args) => {
  return (
    <Pagination>
      <Pagination.Button value="prev" />
      <Pagination.Button {...args} />
      <Pagination.Button value="2" />
      <Pagination.Button value="next" />
    </Pagination>
  );
};

export const PaginationDisabledTemplate = (args) => {
  return (
    <Pagination>
      <Pagination.Button {...args} />
      <Pagination.Button value="1" />
      <Pagination.Button value="2" />
      <Pagination.Button value="next" />
    </Pagination>
  );
};

# Pagination

using the `active` and `disabled` props it is possible to set the enhanced values of the `paginationButton`

## Active

Use to indicate the current page that the user is on

<Canvas>
  <Story name="Active" args={{ active: true, disabled: false, value: '1' }}>
    {PaginationActiveTemplate.bind({})}
  </Story>
</Canvas>

## Disabled

Use to hide unnecessary navigation, for example a "Previous" button if the user is on "Page 1" of a content section

<Canvas>
  <Story
    name="Disabled"
    args={{ active: false, disabled: true, value: 'prev' }}
  >
    {PaginationDisabledTemplate.bind({})}
  </Story>
</Canvas>
